<script>

const PASSWORD="1234"

let friends=[]

function login(){

let p=document.getElementById("pass").value

if(p===PASSWORD){

document.getElementById("login").style.display="none"

}else{

document.getElementById("msg").innerText="كلمة السر خطأ"

}

}

function startBot(){

document.getElementById("status").innerText="الحالة: يعمل"

}

function stopBot(){

document.getElementById("status").innerText="الحالة: متوقف"

}

function addFriend(){

let id=document.getElementById("friendId").value

if(!id)return

friends.push(id)

render()

}

function render(){

let box=document.getElementById("friends")

box.innerHTML=""

friends.forEach((f,i)=>{

let div=document.createElement("div")

div.className="friend"

div.innerHTML=f+' <button class="delete" onclick="del('+i+')">حذف</button>'

box.appendChild(div)

})

}

function del(i){

friends.splice(i,1)

render()

}
